Top 50 Excel Formulas

Top 50 Excel Formulas
1. SUM: =SUM(A1:A10)
2. AVERAGE: =AVERAGE(B1:B10)
3. COUNT: =COUNT(C1:C10)
4. COUNTA: =COUNTA(D1:D10)
5. COUNTIF: =COUNTIF(E1:E10, "criteria")
6. IF: =IF(A1 > 100, "Yes", "No")
7. SUMIF: =SUMIF(F1:F10, "criteria", G1:G10)
8. VLOOKUP: =VLOOKUP(H1, A1:D10, 2, FALSE)
9. HLOOKUP: =HLOOKUP(I1, A1:D10, 2, FALSE)
10. MATCH: =MATCH(J1, A1:A10, 0)
11. INDEX: =INDEX(A1:C10, 3, 2)
12. CONCATENATE: =CONCATENATE("Hello", " ", "World")
13. LEFT: =LEFT(A1, 5)
14. RIGHT: =RIGHT(A1, 5)
15. MID: =MID(A1, 3, 5)
16. LEN: =LEN(A1)
17. FIND: =FIND("text", A1)
18. SEARCH: =SEARCH("text", A1)
19. REPLACE: =REPLACE(A1, 1, 3, "New")
20. SUBSTITUTE: =SUBSTITUTE(A1, "old", "new")
21. TRIM: =TRIM(A1)
22. UPPER: =UPPER(A1)
23. LOWER: =LOWER(A1)
24. PROPER: =PROPER(A1)
25. ROUND: =ROUND(A1, 2)
26. ROUNDDOWN: =ROUNDDOWN(A1, 2)
27. ROUNDUP: =ROUNDUP(A1, 2)
28. INT: =INT(A1)
29. ABS: =ABS(A1)
30. POWER: =POWER(A1, 3)
31. SQRT: =SQRT(A1)
32. TODAY: =TODAY()
33. NOW: =NOW()
34. DAY: =DAY(A1)
35. MONTH: =MONTH(A1)
36. YEAR: =YEAR(A1)
37. HOUR: =HOUR(A1)
38. MINUTE: =MINUTE(A1)
39. SECOND: =SECOND(A1)
40. DATEDIF: =DATEDIF(A1, B1, "d")
41. EOMONTH: =EOMONTH(A1, 1)
42. WORKDAY: =WORKDAY(A1, 5)
43. NETWORKDAYS: =NETWORKDAYS(A1, B1)
44. TEXT: =TEXT(A1, "dd/mm/yyyy")
45. CHOOSE: =CHOOSE(2, "A", "B", "C")
46. ISERROR: =ISERROR(A1/B1)
47. ISBLANK: =ISBLANK(A1)
48. INDIRECT: =INDIRECT("A" & B1)
49. ROW: =ROW(A1)
50. COLUMN: =COLUMN(A1)
